301 virtual void compile() {
302 graph::input_nodes<typename DISPERSION_FUNCTION::base,
303 DISPERSION_FUNCTION::safe_math> inputs = {
304 graph::variable_cast(this->t),
305 graph::variable_cast(this->w),
306 graph::variable_cast(this->x),
307 graph::variable_cast(this->y),
308 graph::variable_cast(this->z),
309 graph::variable_cast(this->kx),
310 graph::variable_cast(this->ky),
311 graph::variable_cast(this->kz)
312 };
313
314 graph::output_nodes<typename DISPERSION_FUNCTION::base,
315 DISPERSION_FUNCTION::safe_math> outputs = {
316 this->residule
317 };
318
319 graph::map_nodes<typename DISPERSION_FUNCTION::base,
320 DISPERSION_FUNCTION::safe_math> setters = {
321 {this->kx_next, graph::variable_cast(this->kx)},
322 {this->ky_next, graph::variable_cast(this->ky)},
323 {this->kz_next, graph::variable_cast(this->kz)},
324 {this->x_next, graph::variable_cast(this->x)},
325 {this->y_next, graph::variable_cast(this->y)},
326 {this->z_next, graph::variable_cast(this->z)},
327 {this->t_next, graph::variable_cast(this->t)}
328 };
329
330 work.add_item(inputs, outputs, setters,
331 graph::shared_random_state<typename DISPERSION_FUNCTION::base,
332 DISPERSION_FUNCTION::safe_math> (),
333 "solver_kernel", inputs.back()->size());
334 work.compile();
335
336 dataset.create_variable(file, "time", this->t, work.get_context());
337 dataset.create_variable(file, "residule", residule, work.get_context());
338 dataset.create_variable(file, "w", this->w, work.get_context());
339 dataset.create_variable(file, "x", this->x, work.get_context());
340 dataset.create_variable(file, "y", this->y, work.get_context());
341 dataset.create_variable(file, "z", this->z, work.get_context());
342 dataset.create_variable(file, "kx", this->kx, work.get_context());
343 dataset.create_variable(file, "ky", this->ky, work.get_context());
344 dataset.create_variable(file, "kz", this->kz, work.get_context());
345
346 file.end_define_mode();
347 }

637 this->z1 = dt*this->D.get_dzdt();

640 graph::pseudo_variable(this->kx + kx1),
641 graph::pseudo_variable(this->ky + ky1),

643 graph::pseudo_variable(this->x + x1),
644 graph::pseudo_variable(this->y + y1),

653 this->y2 = dt*D2.get_dydt();

657 this->ky_next = this->ky + (this->ky1 + this->ky2)/2.0;
658 this->kz_next = this->kz + (this->kz1 + this->kz2)/2.0;

660 this->y_next = this->y + (this->y1 + this->y2 )/2.0;
661 this->z_next = this->z + (this->z1 + this->z2 )/2.0;
662 this->t_next = this->t + dt;
663 }
