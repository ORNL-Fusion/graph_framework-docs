432 template<jit::float_scalar T, bool SAFE_MATH=false>
433 class bohm_gross final : public physics<T, SAFE_MATH> {
434 public:
435//------------------------------------------------------------------------------
451//------------------------------------------------------------------------------
452 virtual graph::shared_leaf<T, SAFE_MATH>
453 D(graph::shared_leaf<T, SAFE_MATH> w,
454 graph::shared_leaf<T, SAFE_MATH> kx,
455 graph::shared_leaf<T, SAFE_MATH> ky,
456 graph::shared_leaf<T, SAFE_MATH> kz,
457 graph::shared_leaf<T, SAFE_MATH> x,
458 graph::shared_leaf<T, SAFE_MATH> y,
459 graph::shared_leaf<T, SAFE_MATH> z,
460 graph::shared_leaf<T, SAFE_MATH> t,
461 equilibrium::shared<T, SAFE_MATH> &eq) {
462
463// Equilibrium quantities.
464 auto ne = eq->get_electron_density(x, y, z);
465 auto wpe2 = build_plasma_fequency(ne, physics<T, SAFE_MATH>::q,
466 physics<T, SAFE_MATH>::me,
467 physics<T, SAFE_MATH>::c,
468 physics<T, SAFE_MATH>::epsion0);
469 auto te = eq->get_electron_temperature(x, y, z);
470// 2*1.602176634E-19 to convert eV to J.
471
472 auto vterm2 = static_cast<T> (2.0)*physics<T, SAFE_MATH>::q*te
473 / (physics<T, SAFE_MATH>::me *
474 physics<T, SAFE_MATH>::c *
475 physics<T, SAFE_MATH>::c);
476
477// Wave numbers should be parallel to B if there is a magnetic field. Otherwise
478// B should be zero.
479 auto b_vec = eq->get_magnetic_field(x, y, z);
480 auto k = kx*eq->get_esup1(x, y, z)
481 + ky*eq->get_esup2(x, y, z)
482 + kz*eq->get_esup3(x, y, z);
483 graph::shared_leaf<T, SAFE_MATH> kpara2;
484 if (b_vec->length()->is_match(graph::zero<T, SAFE_MATH> ())) {
485 kpara2 = k->dot(k);
486 } else {
487 auto b_hat = b_vec->unit();
488 auto kpara = b_hat->dot(k);
489 kpara2 = kpara*kpara;
490 }
491
492 return wpe2 + 3.0/2.0*kpara2*vterm2 - w*w;
493 }
494 };

866//------------------------------------------------------------------------------
867 template<jit::float_scalar T, bool SAFE_MATH=false>
868 class cold_plasma : public physics<T, SAFE_MATH> {
869 public:
870//------------------------------------------------------------------------------
906//------------------------------------------------------------------------------
907 virtual graph::shared_leaf<T, SAFE_MATH>
908 D(graph::shared_leaf<T, SAFE_MATH> w,
909 graph::shared_leaf<T, SAFE_MATH> kx,
910 graph::shared_leaf<T, SAFE_MATH> ky,
911 graph::shared_leaf<T, SAFE_MATH> kz,
912 graph::shared_leaf<T, SAFE_MATH> x,
913 graph::shared_leaf<T, SAFE_MATH> y,
914 graph::shared_leaf<T, SAFE_MATH> z,
915 graph::shared_leaf<T, SAFE_MATH> t,
916 equilibrium::shared<T, SAFE_MATH> &eq) {
917// Dielectric terms.
918// Frequencies
919 auto ne = eq->get_electron_density(x, y, z);
920 auto wpe2 = build_plasma_fequency(ne,
921 physics<T, SAFE_MATH>::q,
922 physics<T, SAFE_MATH>::me,
923 physics<T, SAFE_MATH>::c,
924 physics<T, SAFE_MATH>::epsion0);
925 auto b_vec = eq->get_magnetic_field(x, y, z);
926 auto b_len = b_vec->length();
927 auto ec = build_cyclotron_fequency(-physics<T, SAFE_MATH>::q,
928 b_len,
929 physics<T, SAFE_MATH>::me,
930 physics<T, SAFE_MATH>::c);
931
932 auto w2 = w*w;
933 auto denome = 1.0 - ec*ec/w2;
934 auto e11 = 1.0 - (wpe2/w2)/denome;
935 auto e12 = ((ec/w)*(wpe2/w2))/denome;
936 auto e33 = wpe2;
937
938 for (size_t i = 0, ie = eq->get_num_ion_species(); i < ie; i++) {
939 const T mi = eq->get_ion_mass(i);
940 const T charge = static_cast<T> (eq->get_ion_charge(i))
941 * physics<T, SAFE_MATH>::q;
942
943 auto ni = eq->get_ion_density(i, x, y, z);
944 auto wpi2 = build_plasma_fequency(ni, charge, mi,
945 physics<T, SAFE_MATH>::c,
946 physics<T, SAFE_MATH>::epsion0);
947 auto ic = build_cyclotron_fequency(charge, b_len, mi,
948 physics<T, SAFE_MATH>::c);
949
950 auto denomi = 1.0 - ic*ic/w2;
951 e11 = e11 - (wpi2/w2)/denomi;
952 e12 = e12 + ((ic/w)*(wpi2/w2))/denomi;
953 e33 = e33 + wpi2;
954 }
955
956 e12 = -1.0*e12;
957 e33 = 1.0 - e33/w2;
958
959// Wave numbers.
960 auto n = (kx*eq->get_esup1(x, y, z) +
961 ky*eq->get_esup2(x, y, z) +
962 kz*eq->get_esup3(x, y, z))/w;
963 auto b_hat = b_vec->unit();
964
965 auto npara = b_hat->dot(n);
966 auto npara2 = npara*npara;
967 auto nperp = b_hat->cross(n)->length();
968 auto nperp2 = nperp*nperp;
969
970// Determinate matrix elements
971 auto m11 = e11 - npara2;
972 auto m12 = e12;
973 auto m13 = npara*nperp;
974 auto m22 = e11 - npara2 - nperp2;
975 auto m33 = e33 - nperp2;
976
977 return (m11*m22 - m12*m12)*m33 - m22*(m13*m13);
978 }
979 };
