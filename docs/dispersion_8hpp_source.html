545 D(graph::shared_leaf<T, SAFE_MATH> w,

547 graph::shared_leaf<T, SAFE_MATH> ky,
548 graph::shared_leaf<T, SAFE_MATH> kz,
549 graph::shared_leaf<T, SAFE_MATH> x,

552 graph::shared_leaf<T, SAFE_MATH> t,
553 equilibrium::shared<T, SAFE_MATH> &eq) {
554
555// Equilibrium quantities.
556 auto ne = eq->get_electron_density(x, y, z);
557 auto wpe2 = build_plasma_frequency(ne, physics<T, SAFE_MATH>::q,
558 physics<T, SAFE_MATH>::me,
559 physics<T, SAFE_MATH>::c,
560 physics<T, SAFE_MATH>::epsilon0);
561 auto te = eq->get_electron_temperature(x, y, z);
562// 2*1.602176634E-19 to convert eV to J.
563
564 auto vterm2 = static_cast<T> (2.0)*physics<T, SAFE_MATH>::q*te
565 / (physics<T, SAFE_MATH>::me *
566 physics<T, SAFE_MATH>::c *
567 physics<T, SAFE_MATH>::c);
568
569// Wave numbers should be parallel to B if there is a magnetic field. Otherwise
570// B should be zero.
571 auto b_vec = eq->get_magnetic_field(x, y, z);
572 auto k = kx*eq->get_esup1(x, y, z)
573 + ky*eq->get_esup2(x, y, z)
574 + kz*eq->get_esup3(x, y, z);
575 graph::shared_leaf<T, SAFE_MATH> kpara2;
576 if (b_vec->length()->is_match(graph::zero<T, SAFE_MATH> ())) {
577 kpara2 = k->dot(k);
578 } else {
579 auto b_hat = b_vec->unit();
580 auto kpara = b_hat->dot(k);
581 kpara2 = kpara*kpara;
582 }
583
584 return wpe2 + 3.0/2.0*kpara2*vterm2 - w*w;
585 }

1000 D(graph::shared_leaf<T, SAFE_MATH> w,

1006 graph::shared_leaf<T, SAFE_MATH> z,
1007 graph::shared_leaf<T, SAFE_MATH> t,
1008 equilibrium::shared<T, SAFE_MATH> &eq) {

1010// Frequencies
1011 auto ne = eq->get_electron_density(x, y, z);
1012 auto wpe2 = build_plasma_frequency(ne,
1013 physics<T, SAFE_MATH>::q,
1014 physics<T, SAFE_MATH>::me,
1015 physics<T, SAFE_MATH>::c,
1016 physics<T, SAFE_MATH>::epsilon0);
1017 auto b_vec = eq->get_magnetic_field(x, y, z);
1018 auto b_len = b_vec->length();
1019 auto ec = build_cyclotron_frequency(-physics<T, SAFE_MATH>::q,
1020 b_len,
1021 physics<T, SAFE_MATH>::me,
1022 physics<T, SAFE_MATH>::c);
1023
1024 auto w2 = w*w;
1025 auto denome = 1.0 - ec*ec/w2;
1026 auto e11 = 1.0 - (wpe2/w2)/denome;
1027 auto e12 = ((ec/w)*(wpe2/w2))/denome;
1028 auto e33 = wpe2;
1029
1030 for (size_t i = 0, ie = eq->get_num_ion_species(); i < ie; i++) {
1031 const T mi = eq->get_ion_mass(i);
1032 const T charge = static_cast<T> (eq->get_ion_charge(i))
1033 * physics<T, SAFE_MATH>::q;
1034
1035 auto ni = eq->get_ion_density(i, x, y, z);
1036 auto wpi2 = build_plasma_frequency(ni, charge, mi,
1037 physics<T, SAFE_MATH>::c,
1038 physics<T, SAFE_MATH>::epsilon0);
1039 auto ic = build_cyclotron_frequency(charge, b_len, mi,
1040 physics<T, SAFE_MATH>::c);
1041
1042 auto denomi = 1.0 - ic*ic/w2;
1043 e11 = e11 - (wpi2/w2)/denomi;
1044 e12 = e12 + ((ic/w)*(wpi2/w2))/denomi;
1045 e33 = e33 + wpi2;
1046 }
1047
1048 e12 = -1.0*e12;
1049 e33 = 1.0 - e33/w2;
1050
1051// Wave numbers.
1052 auto n = (kx*eq->get_esup1(x, y, z) +
1053 ky*eq->get_esup2(x, y, z) +
1054 kz*eq->get_esup3(x, y, z))/w;

1059 auto nperp = b_hat->cross(n)->length();
1060 auto nperp2 = nperp*nperp;
1061
1062// Determinate matrix elements
1063 auto m11 = e11 - npara2;
1064 auto m12 = e12;
1065 auto m13 = npara*nperp;
1066 auto m22 = e11 - npara2 - nperp2;
1067 auto m33 = e33 - nperp2;
1068
1069 return (m11*m22 - m12*m12)*m33 - m22*(m13*m13);
1070 }
